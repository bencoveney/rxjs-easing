Installation

Install via NPM:

npm install --save rxjs-easing

Usage

Require it to make the functions available:

var Rx = require("rxjs/Rx");
require("rx.easing");

var myObservable = Rx.Observable.linear(0, 500, 2000);

Example

Make a button bounce on click!

var Rx = require("rxjs/Rx");
require("rxjs-easing");

window.addEventListener("load", function() {
    // Get the button.
    var button = document.querySelector(".example");

    // On click...
    button.addEventListener("click", function() {
        // Prevent more clicks until the easing is complete.
        button.disabled = true;
        
        // Bounce in, then reset.
        Rx.Observable.bounceOut(0, 300, 2000)
            .concat(Rx.Observable.of(0).delay(500))
            .subscribe(
                function(offset) { button.style.left = offset + "px"; },
                console.error,
                function() { button.disabled = false; });
    });
});

API: Back

// The value to start from.
var begin = 0;
// The value to end on.
var end = 100;
// The length of the easing in milliseconds.
var duration = 2000;
// The amount to overshoot by.
var overshoot = 2;

// Ease in.
Rx.Observable.backIn(begin, end, duration, overshoot);

// Ease out.
Rx.Observable.backOut(begin, end, duration, overshoot);

// Ease in & out.
Rx.Observable.backInOut(begin, end, duration, overshoot);

API: Elastic

// The value to start from.
var begin = 0;
// The value to end on.
var end = 100;
// The length of the easing in milliseconds.
var duration = 2000;
// The (optional) amplitude - how big the bounces are.
var amplitude = 1;
// The (optional) period in milliseconds - how often the bounces happen.
var period = 750;

// Ease in.
Rx.Observable.elasticIn(begin, end, duration, amplitude, period);

// Ease out.
Rx.Observable.elasticOut(begin, end, duration, amplitude, period);

// Ease in & out.
Rx.Observable.elasticInOut(begin, end, duration, amplitude, period);
